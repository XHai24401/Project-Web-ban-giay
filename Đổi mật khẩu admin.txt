        [HttpPost]
        public ActionResult frmchangePass(string ex_password, string Password)
        {
            var nv = Session["nhanvien"] as NHANVIEN;
            if (nv.MatKhau == ex_password)
            {
                var model = db.NHANVIENs.Find(nv.MaNV);
                model.MatKhau = Password;
                db.SaveChanges();
                Session["nhanvien"] = null;
                TempData["error"] = "Đổi mật khẩu thành công.Bạn vui lòng đăng nhập lại sau khi đổi mật khẩu.";
                return Redirect("/admin/dangnhap/index");
            }
            else
            {
                TempData["error"] = "Mật khẩu cũ không trùng, vui lòng nhập lại.";
                return Redirect("/admin/dangnhap/changepass");
            }

        }

        public ActionResult Logout()
        {
            Session["nhanvien"] = null;
            return Redirect("/admin/dangnhap/index");
        }
